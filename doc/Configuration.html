def hosts
  @hosts||=[ "localhost" ]
  @hosts = [ @hosts ] if @hosts.kind_of?(String)
  raise "invalid hosts" unless @hosts.kind_of?(Array)
  @hosts
end

def username
  return '' unless @username
  return @username if @username =~ /@$/ or @username.empty?
  return "#{@username}@"
end
